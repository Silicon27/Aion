156 template<typename VecAlloc>
157 void *ASTContext::BumpPtrAllocator<VecAlloc>::allocate(const std::size_t size, const std::size_t alignment,
158 const std::size_t size_of_new_slab,
159 const bool reuse_free_slab) {
160 if (reuse_free_slab && !partially_used_slabs.empty()) {
161 for (auto it = partially_used_slabs.begin(); it != partially_used_slabs.end();) {
162 Slab &slab = slabs[*it];
163 if (void *result = slab.allocate(size, alignment)) {
164 if (slab.get_remaining_capacity() == 0) {
165 it = partially_used_slabs.erase(it);
166 }
167 return result;
168 }
169 ++it;
170 }
171 }
172
173 if (void *result = slabs[current_slab_idx].allocate(size, alignment)) {
174 return result;
175 }
176
177 if (slabs[current_slab_idx].get_remaining_capacity() > 0) {
178 partially_used_slabs.push_back(current_slab_idx);
179 }
180
181 // Handle large allocations by creating a dedicated slab.
182 if (size_of_new_slab == 0 && size >= slab_size) {
183 slabs.emplace_back(size + alignment);
184 return slabs.back().allocate(size, alignment);
185 }
186
187 const std::size_t new_slab_size = size_of_new_slab > 0 ? size_of_new_slab : slab_size;
188 slabs.emplace_back(new_slab_size);
189 current_slab_idx = slabs.size() - 1;
190
191 // Implement geometric growth: double the default slab size for future allocations, up to 128MB.
192 if (size_of_new_slab == 0 && slab_size < 128 * 1024 * 1024) {
193 slab_size *= 2;
194 }
195
196 return slabs[current_slab_idx].allocate(size, alignment);
197 }

200 void ASTContext::BumpPtrAllocator<VecAlloc>::reset_slab(std::size_t idx) {
201 Slab &slab = slabs[idx];
202 slab.reset();
203
204 auto it = std::ranges::find(partially_used_slabs, idx);
205 if (it != partially_used_slabs.end()) {
206 partially_used_slabs.erase(it);
207 }
208 partially_used_slabs.insert(partially_used_slabs.begin(), idx);
209 }
