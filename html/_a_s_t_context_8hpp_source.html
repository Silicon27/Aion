24 template<typename T>
25 constexpr T next_power_of_two(T x) {
26 static_assert(std::is_unsigned_v<T>, "T must be unsigned");
27
28 if (x <= 1) return 1;
29
30 --x; // critical to handle exact powers of two
31
32 for (std::size_t i = 1; i < std::numeric_limits<T>::digits; i <<= 1) {
33 x |= x >> i;
34 }
35
36 return x + 1;
37 }

151 explicit StringMap(ASTContext &ctx, std::size_t initial_capacity = 64) : size(0), bytes_used(0), ctx(ctx) {
152 capacity = next_power_of_two(std::max<std::size_t>(initial_capacity, 4));
153 std::size_t bytes = capacity * sizeof(Slot);
154 slots = static_cast<Slot *>(ctx.allocate(bytes, alignof(Slot)));

160 void rehash(size_type new_cap) {
161 new_cap = next_power_of_two(new_cap);
162 if (new_cap <= capacity) return;
163
164 Slot *old_slots = slots;
165 std::size_t old_cap = capacity;
166
167 std::size_t bytes = new_cap * sizeof(Slot);
168 slots = static_cast<Slot *>(ctx.allocate(bytes, alignof(Slot)));
169 capacity = new_cap;
170
171 for (std::size_t i = 0; i < capacity; ++i) {
172 new(&slots[i]) Slot();
173 }
174
175 size = 0;
176 bytes_used = 0;
177
178 for (std::size_t i = 0; i < old_cap; ++i) {
179 if (old_slots[i].occupied) {
180 insert_with_hash(old_slots[i].key, old_slots[i].value, old_slots[i].hash);
181 }
182 }
183 }

270 Value *find_with_hash(std::string_view key, std::size_t h) const {
271 if (capacity == 0) return nullptr;
272 std::size_t idx = h & (capacity - 1);
273 std::size_t start_idx = idx;
274 while (slots[idx].occupied) {
275 if (slots[idx].hash == h && slots[idx].key == key) {
276 return &slots[idx].value;
277 }
278 idx = (idx + 1) & (capacity - 1);
279 if (idx == start_idx) break;
280 }
281 return nullptr;
282 }
283
284 Value *insert_with_hash(std::string_view key, Value value, std::size_t h) {
285 if (is_full()) rehash(capacity * 2);
286
287 std::size_t idx = h & (capacity - 1);
288 while (slots[idx].occupied) {
289 if (slots[idx].hash == h && slots[idx].key == key) {
290 slots[idx].value = value;
291 return &slots[idx].value;
292 }
293 idx = (idx + 1) & (capacity - 1);
294 }
295 slots[idx] = Slot(key, value, h, true);
296 ++size;
297 bytes_used += key.size();
298 return &slots[idx].value;
299 }

359 template<typename VecAlloc>
360 void *ASTContext::BumpPtrAllocator<VecAlloc>::allocate(const std::size_t size, const std::size_t alignment,
361 const std::size_t size_of_new_slab,
362 const bool reuse_free_slab) {
363 if (reuse_free_slab && !partially_used_slabs.empty()) {
364 for (auto it = partially_used_slabs.begin(); it != partially_used_slabs.end();) {
365 Slab &slab = slabs[*it];
366 if (void *result = slab.allocate(size, alignment)) {
367 if (slab.get_remaining_capacity() == 0) {
368 it = partially_used_slabs.erase(it);
369 }
370 return result;
371 }
372 ++it;
373 }
374 }
375
376 if (void *result = slabs[current_slab_idx].allocate(size, alignment)) {
377 return result;
378 }
379
380 if (slabs[current_slab_idx].get_remaining_capacity() > 0) {
381 partially_used_slabs.push_back(current_slab_idx);
382 }
383
384 // Handle large allocations by creating a dedicated slab.
385 if (size_of_new_slab == 0 && size >= slab_size) {
386 slabs.emplace_back(size + alignment);
387 return slabs.back().allocate(size, alignment);
388 }
389
390 const std::size_t new_slab_size = size_of_new_slab > 0 ? size_of_new_slab : slab_size;
391 slabs.emplace_back(new_slab_size);
392 current_slab_idx = slabs.size() - 1;
393
394 // Implement geometric growth: double the default slab size for future allocations, up to 128MB.
395 if (size_of_new_slab == 0 && slab_size < 128 * 1024 * 1024) {
396 slab_size *= 2;
397 }
398
399 return slabs[current_slab_idx].allocate(size, alignment);
400 }
