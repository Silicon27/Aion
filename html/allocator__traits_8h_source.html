13template <typename Alloc>
14struct allocator_traits {
15 using allocator_type = Alloc;
16 using value_type = typename Alloc::value_type;
17
18 using pointer = typename Alloc::pointer;
19 using const_pointer = typename Alloc::const_pointer;
20 using void_pointer = typename Alloc::void_pointer;
21 using const_void_pointer = typename Alloc::const_void_pointer;
22
23 using difference_type = typename Alloc::difference_type;
24 using size_type = typename Alloc::size_type;
25
26 template <typename T>
27 using rebind_alloc = typename Alloc::template rebind<T>::other;
28
29 template <typename T>
30 using rebind_traits = allocator_traits<rebind_alloc<T>>;
31
32 static pointer allocate(allocator_type& alloc, size_type n) {
33 return alloc.allocate(n);
34 }
35
36 static void deallocate(Alloc& a, pointer p, size_type n) {
37 a.deallocate(p, n);
38 }
39
40 template <typename T, typename... Args>
41 static void construct(Alloc& a, T* p, Args&&... args) {
42 ::new(static_cast<void *>(p)) T(forward<Args>(args)...);
43 }
44
45 template <typename T>
46 static void destroy(Alloc& a, T* p) {
47 p->~T();
48 }
49
50 static size_type max_size(const Alloc& a) noexcept {
51 return a.max_size();
52 }
53
54 static Alloc select_on_container_copy_construction(const Alloc& a) {
55 return a;
56 }
