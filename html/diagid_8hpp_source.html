18// ============================================================================
19// Diagnostic Category Sizes
20// ============================================================================
21// Size of each diagnostic category. These define the maximum number of
22// diagnostics that can be defined in each category.
23
24enum DiagnosticSizes : unsigned {
25 DIAG_SIZE_COMMON = 300,
26 DIAG_SIZE_LEXER = 200,
27 DIAG_SIZE_PARSER = 500,
28 DIAG_SIZE_SEMA = 1000,
29 DIAG_SIZE_CODEGEN = 200,
30};
31
32// ============================================================================
33// Diagnostic Category Start Positions
34// ============================================================================
35// Start position for diagnostics in each category.
36
37enum DiagnosticStart : unsigned {
38 DIAG_START_COMMON = 0,
39 DIAG_START_LEXER = DIAG_START_COMMON + DIAG_SIZE_COMMON,
40 DIAG_START_PARSER = DIAG_START_LEXER + DIAG_SIZE_LEXER,
41 DIAG_START_SEMA = DIAG_START_PARSER + DIAG_SIZE_PARSER,
42 DIAG_START_CODEGEN = DIAG_START_SEMA + DIAG_SIZE_SEMA,
43 DIAG_UPPER_LIMIT = DIAG_START_CODEGEN + DIAG_SIZE_CODEGEN,
44};

159// ============================================================================
160// Diagnostic Mapping
161// ============================================================================
162// Used to track how a diagnostic's severity has been modified.
163
164class DiagnosticMapping {
165 Severity severity_ : 4;
166 bool is_user_ : 1; // Set by user (command line)
167 bool is_pragma_ : 1; // Set by pragma
168 bool no_warning_as_error_ : 1;
169 bool no_error_as_fatal_ : 1;
170
171public:
172 static DiagnosticMapping make(Severity sev, bool is_user = false, bool is_pragma = false) {
173 DiagnosticMapping result;
174 result.severity_ = sev;
175 result.is_user_ = is_user;
176 result.is_pragma_ = is_pragma;
177 result.no_warning_as_error_ = false;
178 result.no_error_as_fatal_ = false;
179 return result;
180 }
181
182 Severity get_severity() const { return severity_; }
183 void set_severity(Severity sev) { severity_ = sev; }
184
185 bool is_user() const { return is_user_; }
186 bool is_pragma() const { return is_pragma_; }
187
188 bool has_no_warning_as_error() const { return no_warning_as_error_; }
189 void set_no_warning_as_error(bool val) { no_warning_as_error_ = val; }
190
191 bool has_no_error_as_fatal() const { return no_error_as_fatal_; }
192 void set_no_error_as_fatal(bool val) { no_error_as_fatal_ = val; }
193};
