35struct CharSourceRange {
36 Source_Location begin;
37 Source_Location end;
38 bool is_token_range_ = true; // true = token range, false = char range
39 bool show_underline_ = true;
40 Source_Location caret_location;
41 bool has_caret_location_ = false;
42
43 CharSourceRange() = default;
44 CharSourceRange(Source_Location b, Source_Location e, bool token = true)
45 : begin(b), end(e), is_token_range_(token) {}
46
47 bool is_valid() const;
48 bool is_token_range() const { return is_token_range_; }
49 bool is_char_range() const { return !is_token_range_; }
50 bool show_underline() const { return show_underline_; }
51 bool has_caret_location() const { return has_caret_location_; }
52
53 static CharSourceRange get_token_range(Source_Location b, Source_Location e) {
54 return CharSourceRange(b, e, true);
55 }
56
57 static CharSourceRange get_char_range(Source_Location b, Source_Location e) {
58 return CharSourceRange(b, e, false);
59 }
60};
61
62inline CharSourceRange token_range(Source_Location b, Source_Location e) {
63 return CharSourceRange::get_token_range(b, e);
64}
65
66inline CharSourceRange char_range(Source_Location b, Source_Location e) {
67 return CharSourceRange::get_char_range(b, e);
68}
69
70inline bool range_contains_location(const CharSourceRange& range, Source_Location loc) {
71 if (!range.is_valid() || !loc.is_valid()) {
72 return false;
73 }
74 if (range.begin.file != loc.file || range.end.file != loc.file) {
75 return false;
76 }
77
78 Offset begin = range.begin.offset;
79 Offset end = range.end.offset;
80 if (begin > end) {
81 std::swap(begin, end);
82 }
83 return loc.offset >= begin && loc.offset <= end;
84}
