30// ============================================================================
31// Source Range (for highlighting)
32// ============================================================================
33
34struct CharSourceRange {
35 Source_Location begin;
36 Source_Location end;
37 bool is_token_range_ = true; // true = token range, false = char range
38
39 CharSourceRange() = default;
40 CharSourceRange(Source_Location b, Source_Location e, bool token = true)
41 : begin(b), end(e), is_token_range_(token) {}
42
43 bool is_valid() const;
44 bool is_token_range() const { return is_token_range_; }
45 bool is_char_range() const { return !is_token_range_; }
46
47 static CharSourceRange get_token_range(Source_Location b, Source_Location e) {
48 return CharSourceRange(b, e, true);
49 }
50
51 static CharSourceRange get_char_range(Source_Location b, Source_Location e) {
52 return CharSourceRange(b, e, false);
53 }
54};
