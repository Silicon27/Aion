123 template <char... Cs>
124 consteval auto remove_trailing_null() {
125 if constexpr (sizeof...(Cs) == 0) {
126 return string<>{};
127 } else {
128 constexpr char arr[] = {Cs...};
129 if constexpr (arr[sizeof...(Cs) - 1] == '\0') {
130 return []<std::size_t... Is>(std::index_sequence<Is...>) {
131 return string<arr[Is]...>{};
132 }(std::make_index_sequence<sizeof...(Cs) - 1>{});
133 } else {
134 return string<Cs...>{};
135 }
136 }
137 }
