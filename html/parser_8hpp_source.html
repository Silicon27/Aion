136 // left binding power: how strongly this token binds to whatever is to its left.
137 static int lbp(const TokenType token) {
138 switch (token) {
139 case TokenType::equal:
140 return 10; // assignment
141
142 case TokenType::double_dot:
143 case TokenType::triple_dot:
144 return 20; // ranges
145
146 case TokenType::equal_equal:
147 case TokenType::bang_equal:
148 return 30; // equality
149
150 case TokenType::less:
151 case TokenType::less_equal:
152 case TokenType::greater:
153 case TokenType::greater_equal:
154 return 40; // comparisons
155
156 case TokenType::plus:
157 case TokenType::minus:
158 return 50; // additive
159
160 case TokenType::star:
161 case TokenType::slash:
162 return 60; // multiplicative
163
164 case TokenType::lparen:
165 return 70; // function call — binds tighter than arithmetic, looser than ::
166 case TokenType::bang:
167 return 75; // postfix !
168
169 case TokenType::dot:
170 case TokenType::double_colon:
171 return 80; // member/scope access
172
173 default:
174 return 0; // not an infix operator (Pratt: no binding power)
175 }
