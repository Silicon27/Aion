34 class BuiltinType : public Type {
35 friend class ASTContext;
36 public:
37 static BuiltinTypeKind get_kind(const lexer::TokenType type) {
38 switch (type) {
39 case lexer::TokenType::kw_i4: return BuiltinTypeKind::i4;
40 case lexer::TokenType::kw_i8: return BuiltinTypeKind::i8;
41 case lexer::TokenType::kw_i16: return BuiltinTypeKind::i16;
42 case lexer::TokenType::kw_i32: return BuiltinTypeKind::i32;
43 case lexer::TokenType::kw_i64: return BuiltinTypeKind::i64;
44 case lexer::TokenType::kw_i128: return BuiltinTypeKind::i128;
45 case lexer::TokenType::kw_f4: return BuiltinTypeKind::f4;
46 case lexer::TokenType::kw_f8: return BuiltinTypeKind::f8;
47 case lexer::TokenType::kw_f16: return BuiltinTypeKind::f16;
48 case lexer::TokenType::kw_f32: return BuiltinTypeKind::f32;
49 case lexer::TokenType::kw_f64: return BuiltinTypeKind::f64;
50 case lexer::TokenType::kw_f128: return BuiltinTypeKind::f128;
51 case lexer::TokenType::kw_char: return BuiltinTypeKind::char_;
52 case lexer::TokenType::kw_bool: return BuiltinTypeKind::bool_;
53 case lexer::TokenType::int_literal: return BuiltinTypeKind::i32; // TODO make this platform dependent, make it the size of the platforms word size
54 case lexer::TokenType::float_literal: return BuiltinTypeKind::f32; // todo this too
55 case lexer::TokenType::string_literal: return BuiltinTypeKind::string_literal;
56 default:
57 // This should not happen if is_builtin_type_token(type) is true
58 return BuiltinTypeKind::i32;
59 }
60 }
